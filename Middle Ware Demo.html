Usage example
const express = require('express');
const expressCrypto = require('express-crypto');

const app = express();

// Register middleware with a shared secret
app.use(expressCrypto({ secret: 'my-32-byte-secret-key' }));

// Route sees plain JSON — no crypto code here
app.post('/api/data', (req, res) => {
  console.log(req.body); // { userId: 42, action: "getData" }
  res.json({ result: 'Hello!' }); // auto-encrypted by middleware
});
